def compress_dataframe(df):
    '''å°†æ‰€æœ‰æ•°æ®çš„ç±»å‹éƒ½è½¬æ¢ä¸ºæ•°å€¼å‹'''
    result = df.copy()
    for col in result.columns:
        col_data = result[col]
        dn = col_data.dtype.name
        if dn == "object":
            result[col] = pd.to_numeric(col_data.astype("category").cat.codes, downcast="integer")
        elif dn == "bool":
            result[col] = col_data.astype("int8")
        elif dn.startswith("int") or (col_data.round() == col_data).all():
            result[col] = pd.to_numeric(col_data, downcast="integer")
        else:
            result[col] = pd.to_numeric(col_data, downcast='float')
    return result

def weather_set_time(df,time_zone):

    GMT_offset_map = {site: offset for site, offset in enumerate(site_GMT_offsets)}

    #æ ¹æ®æ—¶åŒºçš„ä¸åŒï¼Œç»Ÿä¸€æ—¶é—´
    site_dfs = []

def make_is_bad_zero(Xy_subset, min_interval=48, summer_start=3000, summer_end=7500):
    #å¤å¤©ï¼Œ3000/24=125ï¼Œ7500/24=312.5,ç¬¬125å¤©åˆ°ç¬¬312.5å¤©ä¸ºå¤å¤©ã€‚

    meter = Xy_subset.meter_id.iloc[0]
    is_zero = Xy_subset.meter_reading == 0 #è¿”å›è¯»æ•°ä¸º0çš„ç”µè¡¨çš„indices
    if meter == 0:
        #ç”µè¡¨çš„åº¦æ•°ä¸åº”è¯¥ä¸º0ï¼Œæ‰€ä»¥ç”µè¡¨ï¼ˆmeterä¸º0ï¼‰è¯»æ•°ä¸º0çš„è¡Œä»training dataframeä¸­dropæ‰
        return is_zero

    transitions = (is_zero != is_zero.shift(1))#å‡ºç°0å’Œé0å˜åŒ–çš„ä½ç½®
    all_sequence_ids = transitions.cumsum()#åˆ°å„ä½ç½®å‡ºç°çš„å˜åŒ–çš„å’Œï¼Œæ˜¯ä¸€ä¸ªpd.Seires
    ids = all_sequence_ids[is_zero].rename("ids")#å°†å…¶ä¸­è¯»æ•°ä¸º0çš„æå–å‡ºæ¥
    if meter in [2, 3]:
        # è’¸æ±½å’Œçƒ­æ°´æœ‰å¯èƒ½åœ¨å¤å¤©è¢«å…³é—­
        keep = set(ids[(Xy_subset.timestamp < summer_start) |
                       (Xy_subset.timestamp > summer_end)].unique())#ä¸åœ¨å¤å¤©çš„indices
        is_bad = ids.isin(keep) & (ids.map(ids.value_counts()) >= min_interval) 
        #å°†ä¸åœ¨å¤å¤©å´è¢«å…³é—­çš„è’¸æ±½å’Œçƒ­æ°´è¡¨æå–å‡ºæ¥ï¼Œè‡³å°‘è¢«å…³é—­äº†48å°æ—¶ä»¥ä¸Šçš„
    elif meter == 1:
        time_ids = ids.to_frame().join(Xy_subset.timestamp).set_index("timestamp").ids#å°†idså’Œtimestampå¯¹åº”èµ·æ¥
        is_bad = ids.map(ids.value_counts()) >= min_interval#å…³é—­æ—¶é—´å¤§äº48å°æ—¶çš„

        # å†·æ°´åœ¨å†¬å¤©å¯èƒ½è¢«å…³é—­
        jan_id = time_ids.get(0, False)#ä¸€æœˆä»½çš„å¼€å§‹çš„id
        dec_id = time_ids.get(8283, False)#åäºŒæœˆä»½å¼€å§‹çš„id
        if (jan_id and dec_id and jan_id == time_ids.get(500, False) and
                dec_id == time_ids.get(8783, False)):
        #å¦‚æœä¸€æœˆ500å°æ—¶å’ŒåäºŒæœˆ500å°æ—¶çš„è¯»è¡¨éƒ½ä¸º0çš„è¯
            is_bad = is_bad & (~(ids.isin(set([jan_id, dec_id]))))
            #å°†è¿™ä¸€éƒ¨åˆ†çš„çš„è¡Œä»is_badä¸­åˆ é™¤
    else:
        raise Exception(f"Unexpected meter type: {meter}")

    result = is_zero.copy()
    result.update(is_bad)
    return result
